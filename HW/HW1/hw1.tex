\subsection*{Q3-a}
\begin{align*}
\text{Want to proof }
\frac{d}{dt} (f + g) = \frac{df}{dt} + \frac{dg}{dt}
\end{align*}

\fbox{\parbox{\textwidth}{%
\text{On LHS, by definition of differentiation, }
\begin{align*}
\frac{d}{dt} (f + g) &= \lim_{h \to 0} \frac{f(t + h) + g(t + h) - (f(t) + g(t))}{h} \\
&= \lim_{h \to 0} (\frac{f(t + h) - f(t)}{h} + \frac{g(t + h) - g(t)}{h})
\end{align*}
By definition of differentiation, 
\begin{align*}
 \lim_{h \to 0} \frac{f(t + h) - f(t)}{h} &= \frac{df}{dt} \\
 \lim_{h \to 0} \frac{g(t + h) - g(t)}{h} &= \frac{dg}{dt}
\end{align*}
This proved that
\begin{align*}
\frac{d}{dt} (f + g) &= \frac{df}{dt} + \frac{dg}{dt}
\end{align*}
}}
